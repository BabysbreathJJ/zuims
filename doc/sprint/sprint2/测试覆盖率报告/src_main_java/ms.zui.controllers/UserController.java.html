package ms.zui.controllers;

import ms.zui.datamodel.dto.RegisterResultDTO;
import ms.zui.datamodel.dto.UserDTO;
import ms.zui.datamodel.parameter.UserLoginParameterContext;
import ms.zui.datamodel.parameter.UserRegisterParameterContext;
import ms.zui.services.impl.UserServiceImpl;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;
import org.springframework.web.bind.annotation.ResponseBody;
import org.springframework.web.bind.annotation.RestController;

/**
 * 用户操作控制器
 * 
 * @author gaoce
 * @version $Id: UserController.java, v 0.1 2015年10月28日 下午6:58:58 gaoce Exp $
 */
@RestController
public class UserController extends BaseController {

    /** 用户操作层 */
    @Autowired
    private UserServiceImpl userService;

    /**
     * 根据id查询用户
     * 
     * @param email
     * @return user
     */
    @RequestMapping(value = "/users", method = RequestMethod.GET)
    public @ResponseBody UserDTO getUsers(String phone) {
        logger.info("[用户查询接口-根据phone查询]-phone: " + phone);
        UserDTO userDto = userService.getUserInfoByPhone(phone);
        if (userDto == null) {
            throw new IllegalArgumentException("无此用户");
        }
        return userDto;
    }

    @RequestMapping(value = "/users/login", method = RequestMethod.POST)
    public @ResponseBody UserDTO login(@RequestBody UserLoginParameterContext userLoginParameterContext) {
        logger.info("[用户登录接口]-phone: " + userLoginParameterContext.getPhone() + ", password: "
                    + userLoginParameterContext.getPassword());

        UserDTO userDto = userService.login(userLoginParameterContext.getPhone(),
            userLoginParameterContext.getPassword());
        if (userDto == null) {
            throw new IllegalArgumentException("用户名或密码错误");
        }
        return userDto;
    }

    @RequestMapping(value = "/users/register", method = RequestMethod.POST)
    public @ResponseBody RegisterResultDTO register(@RequestBody UserRegisterParameterContext userRegisterParameterContext) {
        logger.info("[用户注册接口]-phone: " + userRegisterParameterContext.getPhone() + ", password: "
                    + userRegisterParameterContext.getPassword() + ", gender: "
                    + userRegisterParameterContext.isMan() + ", last name: "
                    + userRegisterParameterContext.getLastname());
        boolean isSuccess = userService.register(userRegisterParameterContext);
        if (isSuccess == false) {
            throw new RuntimeException("用户已被注册");
        }
        return new RegisterResultDTO(isSuccess);
    }
}
