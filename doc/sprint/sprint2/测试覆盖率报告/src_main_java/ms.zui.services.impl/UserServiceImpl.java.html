package ms.zui.services.impl;

import java.sql.Date;
import java.util.Calendar;

import ms.zui.datamodel.dao.User2UserTypeDAO;
import ms.zui.datamodel.dao.User2VipLevelDAO;
import ms.zui.datamodel.dao.UserDAO;
import ms.zui.datamodel.domain.User;
import ms.zui.datamodel.domain.User2UserType;
import ms.zui.datamodel.domain.User2VipLevel;
import ms.zui.datamodel.dto.UserDTO;
import ms.zui.datamodel.parameter.UserRegisterParameterContext;
import ms.zui.services.UserService;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import util.Constrain;
import util.ConvertTool;

@Service
public class UserServiceImpl implements UserService {

    @Autowired
    UserDAO          userDao;

    @Autowired
    User2UserTypeDAO user2UserTypeDao;

    @Autowired
    User2VipLevelDAO user2VipLevelDao;

    @Override
    public UserDTO getUserInfoByPhone(String phone) {
        User user = userDao.findByPhoneid(phone);
        return ConvertTool.convertFromUser(user, 1);
    }

    @Override
    public UserDTO login(String phone, String password) {
        User user = userDao.findByPhoneid(phone);
        // 密码错误
        if (!password.equals(user.getPassword())) {
            return null;
        } else {
            return ConvertTool.convertFromUser(user, 1);
        }
    }

    @Override
    public boolean register(UserRegisterParameterContext userRegisterParameterContext) {
        // 手机号已被使用
        if (userDao.findByPhoneid(userRegisterParameterContext.getPhone()) != null) {
            return false;
        }

        // 1. 新建用户
        User user = new User();
        user.setPhoneid(userRegisterParameterContext.getPhone());
        if (userRegisterParameterContext.isMan()) {
            user.setGender(1);
        } else {
            user.setGender(0);
        }
        user.setLastName(userRegisterParameterContext.getLastname());
        user.setPassword(userRegisterParameterContext.getPassword());
        userDao.save(user);

        // 2. 确定用户来源，目前全部为默认来源：注册
        User2UserType user2UserType = new User2UserType();
        user2UserType.setUtid(Constrain.defaultUserType);
        user2UserType.setPhoneid(userRegisterParameterContext.getPhone());
        user2UserTypeDao.save(user2UserType);

        // 3. 确定会员权益，注册情况下目前都为普通用户
        User2VipLevel user2VipLevel = new User2VipLevel();
        user2VipLevel.setVlid(Constrain.normalVipLevel);
        user2VipLevel.setPhoneid(userRegisterParameterContext.getPhone());
        user2VipLevel.setStartDate(new Date(Calendar.getInstance().getTimeInMillis()));
        user2VipLevel.setEndDate(Constrain.getUltimateDate());
        user2VipLevelDao.save(user2VipLevel);

        return true;
    }
}
