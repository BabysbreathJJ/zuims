package ms.zui.datamodel.domain;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.Id;
import javax.persistence.Table;
import javax.validation.constraints.NotNull;

/**
 * User实体类
 * 
 * @author Xin
 * @version $Id: Vip_level.java, v 0.1 
 *
 *CREATE TABLE IF NOT EXISTS `mydb`.`vip_level` (
 * `vlid` INT NOT NULL,
 * `vl_name` VARCHAR(45) NOT NULL,
 * PRIMARY KEY (`vlid`))
 *ENGINE = InnoDB;
 *
 *
 */
@Entity
@Table(name = "Vip_level")
public class VipLevel {
    @Id
    @NotNull
    @Column(name = "vlid")
    private int vlid;

    @Column(name = "vl_name")
    @NotNull
    private String vlName;
    
    @Column(name = "buffet_discount")
    private int buffetDiscount;
    
    @Column(name = "lobby_discount")
    private int lobbyDiscount; 
    
    @Column(name = "lunch_discount")
    private int lunchDiscount; 
    
    @Column(name = "bar_discount")
    private int barDiscount; 
    
    @Column(name = "weekend_discount")
    private int weekendDiscount; 
    
    public VipLevel(){}

    public int getVlid() {
        return vlid;
    }

    public void setVlid(int value) {
        this.vlid = value;
    }

    public String getVlName() {
        return vlName;
    }

    public void setVlName(String value) {
        this.vlName = value;
    }
    
    public int getBuffetDiscount() {
    	return buffetDiscount;
    }
    
    public void setBuffetDiscount(int value) {
    	this.buffetDiscount = value;
    }
    
    public int getLobbyDiscount() {
    	return lobbyDiscount;
    }
    
    public void setLobbyDiscount(int value) {
    	this.lobbyDiscount = value;
    }
    
    public int getLunchDiscount() {
    	return lunchDiscount;
    }
    
    public void setLunchDiscount(int value) {
    	this.lunchDiscount = value;
    }
    
    public int getBarDiscount() {
    	return barDiscount;
    }
    
    public void setBarDiscount(int value) {
    	this.barDiscount = value;
    }
    
    public int getWeekendDiscount() {
    	return weekendDiscount;
    }
    
    public void setWeekendDiscount(int value) {
    	this.weekendDiscount = value;
    }
}
