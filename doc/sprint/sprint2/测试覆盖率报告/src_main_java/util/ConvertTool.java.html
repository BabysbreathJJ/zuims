package util;

import ms.zui.datamodel.domain.User;
import ms.zui.datamodel.dto.UserDTO;

/**
 * 数据转换工具类
 * 
 * @author gaoce
 * @version $Id: ConvertTool.java, v 0.1 2015年11月14日 下午6:55:45 gaoce Exp $
 */
public class ConvertTool {

    /**
     * 将Domain模型转换为DTO传输模型
     * 
     * @param user
     * @param vipLevel
     * @return
     */
    public static UserDTO convertFromUser(User user, int vipLevel) {
        if (user == null) {
            return null;
        }

        UserDTO userDto = new UserDTO();
        userDto.setPhone(user.getPhoneid());
        userDto.setLastname(user.getLastName());
        userDto.setEmail(user.getEmail());
        userDto.setGender(user.getGender());
        userDto.setVipLevel(vipLevel);
        return userDto;
    }
}
