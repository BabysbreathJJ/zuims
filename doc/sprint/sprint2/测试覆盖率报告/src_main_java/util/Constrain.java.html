package util;

import java.sql.Date;
import java.util.Calendar;

/**
 * 常量表
 * 
 * @author gaoce
 * @version $Id: Constrain.java, v 0.1 2015年11月14日 上午10:50:02 gaoce Exp $
 */
public final class Constrain {
    /** 一般用户的vlid */
    public static final int normalVipLevel  = 1;

    /** vip用户的vlid */
    public static final int vipVipLevel     = 2;

    /** 默认用户来源 */
    public static final int defaultUserType = 1;

    /** 无限期限，暂时硬编码为2099-12-12 */
    public static final Date getUltimateDate() {
        Calendar calendar = Calendar.getInstance();
        calendar.set(2099, 12, 12);
        return new Date(calendar.getTimeInMillis());
    }
}
